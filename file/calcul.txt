``` ruby

print "Entre Op√©ration (+, -, *, /) : "

n2 = gets.chomp.to_f

case op
when "+"
  result = n1 + n2
  puts "Calcul #{n1} + #{n2} = #{result}"
when "-"
  result = n1 - n2
  puts "Calcul #{n1} - #{n2} = #{result}"
when "*"
  result = n1 * n2
  puts "Calcul #{n1} * #{n2} = #{result}"
when "/"
  if n2 != 0
    result = n1.to_f / n2
    puts "Calcul #{n1} / #{n2} = #{result}"
  else
    puts "error"
  end
else
  puts "Erreur"
end
```
